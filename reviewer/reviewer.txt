1. Menerapkan Autentikasi dan Otorisasi pada RESTful API
Kamu belum memproteksi endpoint DELETE /authentications (Logout) dengan autentikasi sesuai dengan kriteria di src/api/authentications


import express from 'express';

const routes = (handler) => {
  const router = express.Router();
  
  router.post('/', handler.postAuthenticationHandler);
  router.put('/', handler.putAuthenticationHandler);
  router.delete('/', handler.deleteAuthenticationHandler);
  
  return router;
};

export default routes;
