pm.test('when GET detail job', () => {
    const jobId = pm.environment.get('jobId');
    const port = pm.environment.get('port');
    pm.sendRequest(`http://localhost:${port}/jobs/${jobId}`, (error, response) => {
        if (!error) {
            pm.test('it should response with revised content', () => {
                pm.expect(response.code).to.equal(200);

                const responseJson = response.json();

                pm.expect(responseJson.status).to.equal('success');
                pm.expect(responseJson.data).to.be.an('object');
                pm.expect(responseJson.data.id).to.equal(jobId);
                pm.expect(responseJson.data.title).to.equal('Lead Backend Developer');
            });
        }
    });
});
